import math

AWG = {
"10":5.26,
"12":3.31,
"14":2.08,
"16":1.31,
"18":0.82,
"20":0.52
}

def area_fio(bitola):

    if bitola in AWG:
        return AWG[bitola]

    return 0

def calcular_metros_fio(espiras, ranhuras, diametro):

    comprimento_espira = math.pi * diametro

    total = espiras * ranhuras * comprimento_espira

    return total / 1000
