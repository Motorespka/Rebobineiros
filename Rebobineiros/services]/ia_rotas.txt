import sqlite3
import pandas as pd
import os

DB_FILE = "data/rebobineiros.db"

def criar_banco():
    if not os.path.exists("data"):
        os.makedirs("data")
    conn = sqlite3.connect(DB_FILE)
    c = conn.cursor()
    # Usuários
    c.execute('''
        CREATE TABLE IF NOT EXISTS usuarios(
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            nome TEXT,
            usuario TEXT UNIQUE,
            senha TEXT,
            tipo TEXT
        )
    ''')
    # Motores
    c.execute('''
        CREATE TABLE IF NOT EXISTS motores(
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            marca TEXT,
            potencia_cv TEXT,
            rpm TEXT,
            voltagem TEXT,
            amperagem TEXT,
            polaridade TEXT,
            bobina_principal TEXT,
            fio_principal TEXT,
            passo_p TEXT,
            bobina_auxiliar TEXT,
            fio_auxiliar TEXT,
            passo_a TEXT,
            tipo_ligacao TEXT,
            ligacao_interna TEXT,
            rolamentos TEXT,
            selo_mecanico TEXT,
            eixo_x TEXT,
            eixo_y TEXT,
            capacitor TEXT,
            status TEXT,
            obs TEXT
        )
    ''')
    # OS
    c.execute('''
        CREATE TABLE IF NOT EXISTS os(
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            cliente TEXT,
            motor_id INTEGER,
            servico TEXT,
            pecas TEXT,
            status TEXT,
            responsavel TEXT,
            data TEXT
        )
    ''')
    # Grupos
    c.execute('''
        CREATE TABLE IF NOT EXISTS grupos(
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            nome TEXT,
            token TEXT
        )
    ''')
    # Estoque
    c.execute('''
        CREATE TABLE IF NOT EXISTS estoque(
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            item TEXT,
            min_qt INTEGER,
            max_qt INTEGER,
            atual INTEGER
        )
    ''')
    conn.commit()
    conn.close()

def executar(query, params=(), fetch=False):
    conn = sqlite3.connect(DB_FILE)
    c = conn.cursor()
    c.execute(query, params)
    if fetch:
        resultado = c.fetchall()
    else:
        resultado = None
    conn.commit()
    conn.close()
    return resultado

def carregar_tabela(tabela):
    conn = sqlite3.connect(DB_FILE)
    df = pd.read_sql_query(f"SELECT * FROM {tabela}", conn)
    conn.close()
    return df

def salvar_tabela(df, tabela):
    conn = sqlite3.connect(DB_FILE)
    df.to_sql(tabela, conn, if_exists='replace', index=False)
    conn.close()
