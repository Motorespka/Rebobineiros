import streamlit as st
import pandas as pd
from core.database import carregar, salvar

st.title("➕ Cadastro de Motor")

df = carregar("motores.csv")

marca = st.text_input("Marca")
cv = st.text_input("Potência CV")
rpm = st.text_input("RPM")
voltagem = st.text_input("Voltagem")

fio = st.text_input("Bitola fio")
espiras = st.text_input("Espiras")

if st.button("Salvar"):

    novo = pd.DataFrame([{
        "marca":marca,
        "cv":cv,
        "rpm":rpm,
        "voltagem":voltagem,
        "fio":fio,
        "espiras":espiras
    }])

    df = pd.concat([df,novo], ignore_index=True)

    salvar(df,"motores.csv")

    st.success("Motor cadastrado")
