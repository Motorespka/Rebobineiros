# --- FUNÇÃO DE LEITURA (OCR) ---
def ler_imagem(img):
    try:
        # Extrai o texto da foto
        texto = pytesseract.image_to_string(img, lang='por')
        return texto
    except:
        return "Erro ao processar imagem. Instale o Tesseract OCR."
